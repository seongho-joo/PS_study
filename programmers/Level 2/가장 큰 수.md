#include "bits/stdc++.h"
using namespace std;

bool compare(string &a, string &b) {
    return a + b > b + a;
}

string solution(vector<int> numbers) {
    vector<string> sorted;
    string answer = "";
    
    for (int n : numbers) {
        sorted.push_back(to_string(n));
    }
    
    sort(sorted.begin(), sorted.end(), compare);
    
    for (string s : sorted) {
        answer += s;
    }

    return answer.front() == '0' ? "0" : answer;
}

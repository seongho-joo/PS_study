class Solution {

    static int[][] snail;
    static int max_value = 0;
    static int number = 0, row = 0, col = 0;

    public int[] solution(int n) {
        initialize(n);

        while (true) {
            if (rowIncrease(n--) == max_value) {
                break;
            }
            if (colIncrease(n--) == max_value) {
                break;
            }
            if (rowAndColIncrease(n--) == max_value) {
                break;
            }
        }

        return getAnswerAndReturn();
    }

    private int[] getAnswerAndReturn() {
        int[] ret = new int[max_value];
        int i = 0;
        for (int[] s : snail) {
            for (int n : s) {
                ret[i++] = n;
            }
        }
        return ret;
    }

    void initialize(int n) {
        max_value = (n * (n + 1) / 2);
        snail = new int[n][];
        for (int i = 0; i < n; i++) {
            snail[i] = new int[i + 1];
        }
    }

    int rowIncrease(int n) {
        for (int i = 0; i < n; i++) {
            snail[row++][col] = ++number;
        }
        row -= 1;
        col += 1;
        return number;
    }

    int colIncrease(int n) {
        for (int i = 0; i < n; i++) {
            snail[row][col++] = ++number;
        }
        row -= 1;
        col -= 2;
        return number;
    }

    int rowAndColIncrease(int n) {
        for (int i = 0; i < n; i++) {
            snail[row--][col--] = ++number;
        }
        row += 2;
        col += 1;
        return number;
    }
}

#include <string>
#include <vector>
#include <cmath>
#include <algorithm>

using namespace std;

int solution(int n) {
    int answer = 0;
    vector<int> ternary;
    
    while(n != 0) {
        ternary.push_back(n % 3);
        n /= 3;
    }
    
    int exponential = (int)ternary.size() - 1;
    
    for(int i : ternary) {
        answer += (pow(3, exponential--) * i);
    }
    
    return answer;
}
